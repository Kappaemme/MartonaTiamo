    <div id="container">
        <div id="message">So che a volte sbaglio ...</div>
        <button onclick="nextMessage()">Continua</button>
    </div>

    <script>
        // Messaggi da visualizzare
        const messages = [
            "So che a volte sbaglio ...",
            "Ma Ã¨ la prima volta anche per me ..."
        ];
        let currentMessage = 0;

        // Mostra il primo messaggio all'inizio
        document.getElementById('container').style.display = 'block';

        function nextMessage() {
            currentMessage++;
            if (currentMessage < messages.length) {
                document.getElementById('message').innerText = messages[currentMessage];
                document.getElementById('message').style.animation = 'fadeIn 2s ease-in-out'; // Riapplica l'animazione
            } else {
                document.getElementById('message').innerText = ""; // Nasconde il messaggio finale
            }
        }
    </script>
